def commentize(description, indent: '')
  lines = description.split("\n")

  comment = ''.dup

  lines.each do |line|
    if line == ''
      comment << "#{indent}#\n"
    else
      sublines = break_line(line, 120 - indent.length)
      sublines.each do |subline|
        comment << "#{indent}# #{subline}\n"
      end
    end
  end

  comment
end
